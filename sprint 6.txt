        document.addEventListener('DOMContentLoaded', function() {
            // Mostrar/ocultar detalles de transacción
            const verDetalleButtons = document.querySelectorAll('.ver-detalle');
            const transactionDetail = document.getElementById('transaction-detail');
            const closeDetailButton = document.getElementById('close-detail');
            
            verDetalleButtons.forEach(button => {
                button.addEventListener('click', function() {
                    const row = this.closest('tr');
                    const cells = row.querySelectorAll('td');
                    
                    // Actualizar información en el detalle
                    document.getElementById('detail-id').textContent = cells[0].textContent;
                    document.getElementById('detail-client').textContent = cells[1].textContent;
                    document.getElementById('detail-product').textContent = cells[2].textContent;
                    document.getElementById('detail-date').textContent = cells[3].textContent;
                    
                    // Actualizar estado
                    const statusElement = document.getElementById('detail-status');
                    statusElement.textContent = cells[4].textContent;
                    statusElement.className = 'status ' + cells[4].querySelector('.status').className;
                    
                    // Mostrar el panel de detalles
                    transactionDetail.style.display = 'block';
                    
                    // Desplazar hacia el panel de detalles
                    transactionDetail.scrollIntoView({ behavior: 'smooth' });
                });
            });
            
            closeDetailButton.addEventListener('click', function() {
                transactionDetail.style.display = 'none';
            });
            
            // Simular filtros
            const statusFilter = document.getElementById('status-filter');
